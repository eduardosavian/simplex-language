// Builtin rars syscall wrappers, leave the implementation empty //
func __write_int(_: int){}
func __write_string(_: string){}
func __read_int() -> int {}
func __write_newline(){}
///////////////////////////////////////////////////////////////////


n: int = 0;

for n < 10 {
	ok: bool = false;
	// Is even?

	if n % 3 == 0 {
		__write_string("Fizz");
		ok = true;
	}
	if n % 5 == 0 {
		__write_string("Buzz");
		ok = true;
	}
	if !ok {
		__write_int(n);
	}

	n = n + 1;
}

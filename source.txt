// Builtin rars syscall wrappers, leave the implementation empty //
func __write_int(_: int){}
func __write_string(_: string){}
func __read_int() -> int {}
func __write_newline(){}
///////////////////////////////////////////////////////////////////


n: int = 1;

for n < 30 {
	ok: int = 0;
	// Is even?
	if n % 3 == 0 {
		__write_string("Fizz");
		ok = 1;
	}
	if n % 5 == 0 {
		__write_string("Buzz");
		ok = 1;
	}

	if ok == 0 {
		__write_int(n);
	}
	__write_newline();

	n = n + 1;
}
